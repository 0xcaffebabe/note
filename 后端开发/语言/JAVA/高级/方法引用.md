```java
public class Main {

    public static void main(String[] args) {
        run(System.out::print);
    }

    private static void run(Printable printable) {
        printable.print("abc");
    }
}

interface Printable {
    void print(String str);
}
```

```java
public class Main {

    public static void main(String[] args) {
        System.out.println(getFoo(Foo::new));
    }

    private static Foo getFoo(Builder builder){
        return builder.builder(new Bar());
    }
}

@FunctionalInterface
interface Builder{
    Foo builder(Bar bar);
}

class Foo{
    public Foo(Bar bar){

    }
}

class Bar{}
```
