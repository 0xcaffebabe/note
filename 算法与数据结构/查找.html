int l = 0, r = a.length - 1;
while (l <= r) {
    int mid = l + (r - l) / 2;
    if (a[mid].equals(target)) {
        return mid;
    }
    if (less(target, a[mid])) {
        r = mid - 1;
    } else {
        l = mid + 1;
    }
}
return -1;
